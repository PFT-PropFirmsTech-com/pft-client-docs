```typescript
export const useGetMyCommissions = (query?: {
  page?: number;
  limit?: number;
  tier?: number;
  sortBy?: string;
  sortOrder?: "asc" | "desc";
}) => {
  return useQuery<{ data: MyCommissionEntry[]; meta: { page: number; limit: number; total: number; totalPages: number } }, Error>({
    queryKey: ["myCommissions", query],
    queryFn: async () => {
      const params = new URLSearchParams();
      params.append("page", String(query?.page || 1));
      params.append("limit", String(query?.limit || 20));
      if (query?.tier !== undefined) params.append("tier", String(query.tier));
      if (query?.sortBy) params.append("sortBy", query.sortBy);
      if (query?.sortOrder) params.append("sortOrder", query.sortOrder);
      const url = `${ENDPOINTS.affiliate.myCommissions}?${params.toString()}`;
      const response = await apiClient.get(url);
      if (!response.data.success) throw new Error(response.data.message || "Failed to fetch commissions");
      return {
        data: response.data.data || [],
        meta: response.data.meta || { page: 1, limit: 20, total: 0, totalPages: 0 },
      };
    },
    staleTime: 30_000,
  });
};
```

```typescript
const exportPurchaseReportCsv = async (tier?: number) => {
  try {
    const params = new URLSearchParams();
    params.append("page", "1");
    params.append("limit", "0"); // fetch all
    if (tier !== undefined) params.append("tier", String(tier));
    const url = `${ENDPOINTS.affiliate.myCommissions}?${params.toString()}`;
    const response = await apiClient.get(url);
    const rows: MyCommissionEntry[] = response.data?.data || [];
    if (rows.length === 0) return;
    const escapeField = (val: unknown): string => {
      if (val === null || val === undefined) return "";
      const str = String(val);
      return str.includes(",") || str.includes('"') || str.includes("\n")
        ? `"${str.replace(/"/g, '""')}"` : str;
    };
    const product = (r: MyCommissionEntry) => {
      const ct = r.payment?.challengeType || "";
      const size = r.payment?.accountSize ? `$${Number(r.payment.accountSize).toLocaleString()}` : "";
      return [ct, size].filter(Boolean).join(" ");
    };
    const headers = ["Client Name","Client Email","Trading Account #","Product","Date","Purchase Amount","Commission %","Commission Amount","Commission Currency"];
    const csvRows = rows.map((r) => [
      escapeField(r.buyer?.name ?? ""),
      escapeField(r.buyer?.email ?? ""),
      escapeField(r.payment?.mt5Login ?? "—"),
      escapeField(product(r)),
      escapeField(r.createdAt ? new Date(r.createdAt).toISOString() : ""),
      escapeField(r.payment?.paidAmount ?? r.payment?.totalPrice ?? ""),
      escapeField(r.rate ?? ""),
      escapeField(r.amount ?? ""),
      escapeField(r.currency ?? r.payment?.currency ?? ""),
    ].join(","));
    const csv = [headers.join(","), ...csvRows].join("\n");
    const blob = new Blob(["﻿" + csv], { type: "text/csv;charset=utf-8" });
    saveAs(blob, `purchase-report${tier ? `-tier-${tier}` : ""}-${new Date().toISOString().split("T")[0]}.csv`);
  } catch (err) {
    console.error("Purchase report export failed", err);
  }
};
```
